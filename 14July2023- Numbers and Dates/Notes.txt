1. using primitive type.
2. using Number constructor( by creating an object)

            if (input < 10) {

        }
        setInterval(function () {
            var date = new Date();
            var hms = addZero(date.getHours()) + ":" + addZero(date.getMinutes()) + ":" + addZero(date.getSeconds());
